// Create a method to disable reserved dates

{
    // Get the reserved dates from your data source (e.g., a list of reserved dates)
    List<DateTime> reservedDates = GetReservedDates();

    myDatePicker.BlackoutDates.Clear();

    // Add reserved dates to the blackout dates collection
    foreach (DateTime reservedDate in reservedDates)
    {
        myDatePicker.BlackoutDates.Add(new CalendarDateRange(reservedDate));
    }
}

// Method to retrieve reserved dates from your data source
private List<DateTime> GetReservedDates()

In this example, the DisableReservedDates() method retrieves the reserved dates from a data source (e.g., a list) using the GetReservedDates() method. To disable a specific period, such as from 20/05/2023 13:30 to 20/05/2023 16:30, in a DatePicker control in WPF, you can use the BlackoutDates property along with a custom CalendarDateRange object. Here's an example of how you can achieve that:

csharp
using System;
using System.Collections.Generic;
using System.Collections.ObjectModel;
using System.ComponentModel;
using System.Windows;

namespace YourNamespace
{
    public class DatePickerViewModel : INotifyPropertyChanged
    {
        private ObservableCollection<DateTime> reservedDates;
        public ObservableCollection<DateTime> ReservedDates
        {
            get { return reservedDates; }
            set
            {
                reservedDates = value;
                OnPropertyChanged("ReservedDates");
            }
        }

        private DateTime disabledStartTime;
        public DateTime DisabledStartTime
        {
            get { return disabledStartTime; }
            set
            {
                disabledStartTime = value;
                OnPropertyChanged("DisabledStartTime");
                UpdateDisabledDates();
            }
        }

        private DateTime disabledEndTime;
        public DateTime DisabledEndTime
        {
            get { return disabledEndTime; }
            set
            {
                disabledEndTime = value;
                OnPropertyChanged("DisabledEndTime");
                UpdateDisabledDates();
            }
        }

        public DatePickerViewModel()
        {
            // Initialize the ReservedDates collection with some sample data
            ReservedDates = new ObservableCollection<DateTime>
            {
                new DateTime(2023, 5, 20, 13, 30, 0),
                new DateTime(2023, 5, 20, 16, 30, 0)
            };

            DisabledStartTime = new DateTime(2023, 5, 20, 13, 30, 0);
            DisabledEndTime = new DateTime(2023, 5, 20, 16, 30, 0);
        }

        public void UpdateDisabledDates()
        {
            Application.Current.Dispatcher.Invoke(() =>
            {
                Application.Current.MainWindow?.FindName("myDatePicker") is DatePicker datePicker &&
                datePicker.BlackoutDates.Clear();

                foreach (DateTime reservedDate in ReservedDates)
                {
                    datePicker.BlackoutDates.Add(new CalendarDateRange(reservedDate));
                }

                datePicker.BlackoutDates.Add(new CalendarDateRange(DisabledStartTime, DisabledEndTime));
            });
        }

        #region INotifyPropertyChanged Implementation

        public event PropertyChangedEventHandler PropertyChanged;

        protected virtual void OnPropertyChanged(string propertyName)
        {
            PropertyChanged?.Invoke(this, new PropertyChangedEventArgs(propertyName));
        }

        #endregion
    }
}
